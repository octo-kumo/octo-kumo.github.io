SQL Database Version Check Cheat Sheet

June 2024 · edited December 2024Created 2024-06-09   Updated 2024-12-151 min read · 43 words
-- MySQL/MariaDB
SELECT VERSION();
-- PostgreSQL
SELECT version();
-- Microsoft SQL Server
SELECT @@VERSION;
-- Oracle Database
SELECT * FROM v$version;
-- SQLite
SELECT sqlite_version();
-- DB2
SELECT service_level, fixpack_num FROM TABLE (sysproc.env_get_inst_info());
-- Informix
SELECT DBINFO('version', 'full');
-- Sybase ASE
SELECT @@version;
