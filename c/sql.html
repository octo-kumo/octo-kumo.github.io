SQL Database Version Check Cheat Sheet

June 2024 · edited December 2024Created 2024-06-09T16:58:00.000Z   Updated 2024-12-15T17:31:00.000Z1 min read43 words
-- MySQL/MariaDB
SELECT VERSION();
-- PostgreSQL
SELECT version();
-- Microsoft SQL Server
SELECT @@VERSION;
-- Oracle Database
SELECT * FROM v$version;
-- SQLite
SELECT sqlite_version();
-- DB2
SELECT service_level, fixpack_num FROM TABLE (sysproc.env_get_inst_info());
-- Informix
SELECT DBINFO('version', 'full');
-- Sybase ASE
SELECT @@version;
