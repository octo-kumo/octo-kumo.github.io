Hardcore method

from pwn import *
context.log_level = 'error'
s = remote('24.199.110.35', 42189)
s.sendlineafter(b'? ', b'36')
for i in range(36):
    m = s.recvregex(br'what is ([0-9]+) \+ ([0-9]+) =', capture=True)
    s.sendline(str(int(m.group(1)) + int(m.group(2))).encode())
    print(m)
print(s.recvall())
s.close()
